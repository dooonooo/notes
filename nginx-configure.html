Nginx配置

2016/1/30 comments

最近看了一本《高性能Linux服务器构建实战》，里面讲到Nginx，感觉以前使用nginx系统配置多不专业，一些要设置，都没有设置。
现在整理一下，基本算是完整了。

基本配置

#user  nobody;
worker_processes  1;
#error_log  logs/error.log;
error_log  logs/error.log  notice;
#error_log  logs/error.log  info;
pid        logs/nginx.pid;
events {
    worker_connections  1024;
}
http {
    include       mime.types;
    default_type  application/octet-stream;
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';
    access_log  logs/access.log  main;
    sendfile        on;

这里用到了include指令，其中/usr/nginx/conf/vhosts/www.domain2.com.conf的内容为： 

server {
    listen 80;
    server_name www.domain2.com;
    access_log logs/domain2.access.log main;
    location / {
        index index.html;
        root html/domain2.com/;
    }
}

防盗链配置

#针对文件名后缀设置防盗链    
location ~* \.(jpg|gif|png|swf|flv|wma|wmv|asf|mp3|mmf|zip|rar)$ {
    valid_referers none blocked *.lidong.me lidong.me;
    if ($invalid_referer) { 
        rewrite ^/ http://www.lidong.me/img/error.gif;
        #return 403;
    }
}
#对目录设置防盗链
location /images {
root html;
    valid_referers none blocked *.lidong.me lidong.me;
    if ($invalid_referer) {
        return 403;
        }
    }
}
