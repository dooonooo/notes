canvas基础

2016/3/19 posted in  canvas comments

HTML

<canvas id="canvas"><canvas>

Javascript

var canvas=document.getElemenById("canvas")
var context=canvas.getContext('2d');

//使用context进行绘制

画直线

//状态设置
context.moveTo(100,100);
context.lineTo(700,700);
context.lineWidth=10;//线条宽度
context.strokeStyle="#058";// 线条颜色
//绘制
context.stroke();

//创建一个全新的路径绘制
context.beginPath();
//关闭路径，绘制封闭的图形
context.closePath();

//颜色填充
fillStyle ="yellow";
fill();

//矩形
rect(x,y,width,height);
//包含填充的矩形
fillRect(x,y width，height);
//绘制边框
strokeRect();

线条样式

//设置线条的开始、结尾显示样式
lineCap="butt"(default)
         "round"（圆形）
         "square"   （方形）
//线条连接处
lineJoin = "miter"(default)尖角
           "bevel"
           "round"圆角           
miterLimit（默认为10）

图形

//图形状态保存
save()
restore()

translate(x,y);  //位移
rotate(deg);  //旋转
scale(sx,sy);   //缩放

//图形变换操作(变换矩阵)
transform(a,b,c,d,e,f);
setTransform(a,b,c,d,e,f);


//填充样式
fillStyle = color
            gradient
            image
            canvas
            video

fillStyle = image||canvas||video
createPattern(img,repeat-style)        
          (canvas,repeat-style)
          (video,repeat-style)

          repeat-style:no-repeat
                        repeat-x
                        repeat-y
                        repeat

线条样式 

strokestyle:

曲线绘制 

context.arc(centerx,centery,radius,startingAngle,endingAngle, anticlockwise = false)
context.arcTo(x1,y1,x2,y2,radius)//沿切线绘制曲线

贝塞尔曲线

context.quadraticCurveTo(
                    x1,y1,   //---控制点
                    x2,y2)   //---结束点
context,bezeirCurveTo(
                    x1,y1,x2,y2,   //---控制点
                    x3,y3)         //---结束点

文字渲染

context.font = "bold  40px Arial"

context.fillText(string,x,y,[maxlen]);
context.strokeText(string,x,y,[maxlen]);
